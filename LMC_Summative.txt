# Valid mnemonics are:

# Get inputs; check if bases > 0 and < 11.
	IN
	STO	num
	IN
	STO	base_a
	SUB	eleven
	BRP	error
	IN
	STO	base_b
	SUB	eleven
	BRP	error
# Convert to base 10
divide	LDA	num
	BRZ	result
	SUB	ten
	BRP	pos
	BR	check_d
pos	STO	num
	LDA	q
	ADD	one
	STO	q
	BR	divide
check_d	LDA	num
	SUB	base_a
	BRP	error
b_ten	LDA	num
	SUB	counter
	BRZ	finish
	LDA	inter
	ADD	posit
	STO	inter
	LDA	counter
	ADD	one
	STO	counter
	BR	b_ten

finish	STO	counter
	LDA	q
	STO	num
	LDA	counter
	STO	q
times	LDA	base_a
	SUB	counter
	BRZ	up_posi
	LDA	q
	ADD	posit
	STO	q
	LDA	counter
	ADD	one
	STO	counter
	BR	times
up_posi	STO	counter
	LDA	q
	STO	posit
	LDA	counter
	STO	q
	BR	divide
	
# Display error message
error	LDA	msg
	BR	end
# Display result
result	LDA	inter
end	OUT
	HLT

num	DAT
base_a	DAT
base_b	DAT
msg	DAT	999
one	DAT	001
ten	DAT	010
eleven	DAT	011
q	DAT	000
counter	DAT	000
